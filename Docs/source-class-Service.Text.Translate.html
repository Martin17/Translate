258:             "text" => $stringToTranslate,
259:             "hl" => $this->translateLang,

261:             "tl" => $this->translateLang,
262:             "ie" => "UTF-8",
263:             "oe" => "UTF-8",

273:         foreach ($params as $key => $value) {
